17  package com.btmatthews.maven.plugins.spoon;

30  import spoon.reflect.Factory;

32  import spoon.support.DefaultCoreFactory;
33  import spoon.support.JavaOutputProcessor;

35  import spoon.support.StandardEnvironment;
36  
37  import java.io.File;
38  import java.util.regex.Matcher;
39  import java.util.regex.Pattern;

41  /**
42   * @author <a href="mailto:brian@btmatthews.com">Brian Matthews</a>
43   * @since 1.0.0
44   */
45  @Mojo(
46          name = "spoon",
47          defaultPhase = LifecyclePhase.PROCESS_SOURCES,
48          requiresDependencyResolution = ResolutionScope.COMPILE_PLUS_RUNTIME,
49          configurator = "include-project-dependencies")
50  public class SpoonMojo extends AbstractMojo {
51  
52      /**
53       * The Java compliance level.
54       */
55      @Parameter(property = "maven.compiler.source", defaultValue = "1.5")
56      private String source;
57      /**
58       * A list of folders containing Java source files to be processed.
59       */
60      @Parameter(required = true)
61      private File[] inputSources;
62      /**
63       * A list of Spoon processors. A processor can be a fully qualified class name or the path for a Groovy script.
64       */
65      @Parameter(required = true)
66      private String[] processors;
67      /**
68       * The Maven project descriptor.
69       */
70      @Parameter(defaultValue = "${project.build.directory}/spooned", required = true)
71      private File outputDirectory;
72      /**
73       *
74       */
75      @Parameter(property = "project", required = true, readonly = true)
76      private MavenProject project;
77  
78      /**
79       * @throws MojoExecutionException
80       */
81      @Override
82      public void execute() throws MojoExecutionException, MojoFailureException {
83          final int complianceLevel = getComplianceLevel();
84          if (complianceLevel >= 5 && complianceLevel <= 8) {
85              try {
86                  outputDirectory.mkdirs();
87                  doExecute(complianceLevel);
88                  project.addCompileSourceRoot(outputDirectory.getAbsolutePath());
89              } catch (final Exception e) {
90                  getLog().error(e.getMessage(), e);
91                  throw new MojoExecutionException(e.getMessage(), e);
92              }
93          } else {
94              final String message = "Invalid or unsupported source level. Must be 1.5, 1.6, 1.7 or 1.8";
95              getLog().error(message);
96              throw new MojoExecutionException(message);
97          }
98      }
99  
100     /**
101      * Parse the source compliance level.
102      *
103      * @return The source compliance level or {@code 0} if the source compliance level could not be parsed.
104      */
105     private int getComplianceLevel() {
106         final Matcher matcher = Pattern.compile("^1\\.([5-8])$").matcher(source);
107         if (matcher.matches()) {
108             return Integer.valueOf(matcher.group(1));
109         } else {
110             return 0;
111         }
112     }
113 
114     /**
115      * @param complianceLevel
116      * @throws Exception
117      */
118     private void doExecute(final int complianceLevel) throws Exception {
119 
120         getLog().info("Source compliance level: 1." + complianceLevel);
121         getLog().info("Write processed sources to: " + outputDirectory.getAbsolutePath());
122 
123         final StandardEnvironment env = new StandardEnvironment();
124         env.setVerbose(false);
125         env.setDebug(false);
126         env.setComplianceLevel(complianceLevel);
127         final Factory factory = new Factory(new DefaultCoreFactory(), env);
128 
129         final Builder builder = factory.getBuilder();
130         for (final File inputSource : inputSources) {
131             getLog().info("Adding input source: " + inputSource.getPath());
132             builder.addInputSource(inputSource);
133         }
134         builder.build();
135 
136         final ProcessingManager processing = new QueueProcessingManager(factory);
137         for (final String processor : processors) {
138             getLog().info("Adding processor: " + processor);
139             if (processor.endsWith(".groovy")) {
140                 final File script = new File(processor);
141                 if (script.exists()) {
142                     processing.addProcessor(new GroovyProcessor<CtElement>(script));
143                 }
144             } else {
145                 processing.addProcessor(processor);
146             }
147         }
148         processing.addProcessor(new JavaOutputProcessor(outputDirectory));
149         processing.process();
150     }
