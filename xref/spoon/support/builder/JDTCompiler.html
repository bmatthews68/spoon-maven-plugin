55          public CompilationUnitDeclaration[] compileUnits(
56                  CompilationUnit[] sourceUnits) {
57              CompilationUnitDeclaration unit = null;
58              int i = 0;
59              // build and record parsed units
60              beginToCompile(sourceUnits);
61              // process all units (some more could be injected in the loop by
62              // the lookup environment)
63              for (; i < this.totalUnits; i++) {
64                  unit = unitsToProcess[i];
65                  this.parser.getMethodBodies(unit);
66  
67                  // fault in fields & methods
68                  if (unit.scope != null)
69                      unit.scope.faultInTypes();
70                  // verify inherited methods
71                  if (unit.scope != null)
72                      unit.scope
73                              .verifyMethods(lookupEnvironment.methodVerifier());
74                  // type checking
75                  unit.resolve();
76                  // flow analysis
77                  unit.analyseCode();
78  
79                  requestor.acceptResult(unit.compilationResult.tagAsAccepted());
80              }
81              return this.unitsToProcess;
82          }

177     /*
178      * Build the set of compilation source units
179      */
180     public CompilationUnit[] getCompilationUnits(List<CtFile> streams)
181             throws Exception {
182         CompilationUnit[] units = new CompilationUnit[streams.size()];
183         int i = 0;
184         for (CtFile stream : streams) {
185             InputStream in = stream.getContent();
186             units[i] = new CompilationUnit(Util.getInputStreamAsCharArray(in,
187                     -1, null), stream.getPath(), null);
188             in.close();
189             i++;
190         }
191         return units;
192     }
